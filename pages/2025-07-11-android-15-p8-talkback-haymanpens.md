	tooltipimg.setAttribute('visibility', 'visible');
      } else {
	console.error(`Element with ID "${tooltipId}" not found.`);
      }
    }

      const tooltipimg = document.getElementById(tooltipId);
      tooltipimg.setAttribute('visibility', 'hidden');
